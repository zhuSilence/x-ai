# 语雀API配置示例
# 请将此文件重命名为 .env 或在环境变量中设置以下配置

# === 语雀API配置 ===
# 语雀API Token，从语雀设置页面获取
# 获取方法：https://www.yuque.com/settings/tokens
YUQUE_TOKEN=your_yuque_token_here

# 知识库命名空间，格式：用户名或团队名/知识库路径
# 例如：personal_user/my_book 或 company_team/tech_docs
YUQUE_NAMESPACE=owner_login/book_slug

# 语雀API基础URL（可选，默认为线上地址）
# 线上地址：https://yuque-api.antfin-inc.com
# 预发地址：https://yuque-pre.antfin-inc.com
YUQUE_BASE_URL=https://yuque-api.antfin-inc.com

# === 发布设置 ===
# 是否启用语雀发布（true/false）
PUBLISH_TO_YUQUE=true

# 文档格式（markdown/html）
# 推荐使用 markdown，格式更清晰
YUQUE_DOC_FORMAT=markdown

# 文档公开性（0-私密，1-公开）
YUQUE_DOC_PUBLIC=0

# === Twitter API配置 ===
# Twitter API Bearer Token
TWITTER_BEARER_TOKEN=your_twitter_bearer_token

# Twitter API等级（free/basic/pro/enterprise）
TWITTER_API_TIER=free

# 安全系数（0.1-1.0，推荐0.8）
TWITTER_SAFETY_FACTOR=0.8

# === 使用说明 ===
# 1. 获取语雀API Token：
#    - 登录语雀
#    - 访问 https://www.yuque.com/settings/tokens
#    - 创建新的Personal Access Token
#    - 复制token到YUQUE_TOKEN配置项
#
# 2. 配置知识库命名空间：
#    - 找到目标知识库的URL，例如：https://yuque.com/company/tech-docs
#    - 提取命名空间：company/tech-docs
#    - 设置YUQUE_NAMESPACE=company/tech-docs
#
# 3. 权限要求：
#    - Token需要有目标知识库的写入权限
#    - 如果是团队知识库，需要确保用户是成员且有权限
#
# 4. 测试连接：
#    - 运行程序时会自动测试API连接和权限
#    - 如果失败会显示具体错误信息
